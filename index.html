Time Series Analysis

  • 24hr (day/night) fluctuations
    • So significant differences in the trend
    • No upwards or downwards movement
  • No specific events other than day/night fluctuations
week <- readRDS("D:/Office/Arbeit/pubs/#justiceforgeorgefloyd/twitter.rds")
week$date <- format(week$created_at, format="%Y%m%d %H")
week$date <- ymd_h(week$date)
date_no <- week %>% count(date)

twitter_df <- merge(week, date_no, all=TRUE)

ggplot(twitter_df, aes(y=n, x=date)) +
  geom_line() +
  ggtitle("Tweets by Hour") +
  xlab("Time") +
  ylab("Number of Tweets") +
  geom_smooth(aes(y=n, x=date))

twitter_df$ts_n <- ts(twitter_df$n, frequency = 24)
decomp <- decompose(twitter_df$ts_n, type="additive")
plot(decomp)

week %>%

  top_n(20) %>%

Most frequent words

twitter_df$text_2  <- tolower(twitter_df$text)
twitter_df$text_2  <-  gsub("amp", "", twitter_df$text_2)
twitter_df$text_2 <-  gsub("https\\s*", "", twitter_df$text_2)
twitter_df$text_2 <-  gsub("@\\S*", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("blacklivesmatter", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("blm", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("[\r\n]", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("[[:punct:]]", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("it's", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("it‘s", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("its", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub(" de ", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("don't", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("don‘t", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("‘", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub("’", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub(" dont ", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub(" its ", "", twitter_df$text_2)
twitter_df$text_2  <-  gsub(" im ", "", twitter_df$text_2)

tweets <- twitter_df %>%
  select(text_2) %>%
  unnest_tokens(word, text_2)

tweets <- tweets %>%
  anti_join(tidytext::stop_words)

tweets %>%
  count(word, sort = TRUE) %>%
  top_n(20) %>%
  mutate(word = reorder(word, n)) %>%
  ggplot(aes(x = word, y = n)) +
  geom_col() +
  xlab(NULL) +
  coord_flip() +
  labs(y = "Count",
       x = "Unique words",
       title = "Most frequent words found in the tweets with BLM",
       subtitle = "Stop words and #blacklivesmatter/blm removed from the list")
